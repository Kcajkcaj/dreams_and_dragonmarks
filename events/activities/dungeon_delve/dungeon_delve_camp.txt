dungeon_delve_camp.0300 = { # Weapon calls out to you
	type = activity_event
	title = camp.0300.t
	desc = camp.0300.desc
	
	theme = dungeon_long_rest
		
	left_portrait = root
	right_portrait = scope:duelling_partner

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = lifestyle_blademaster
		}
	}
	
	immediate = {
		involved_activity = {
			random_attending_character = {
				limit = {not = {root = this}}
				save_scope_as = duelling_partner
			}
		}
	}
	
	option = {
		name = camp.0300.duel
		
		duel = {
			skill = prowess
			target = scope:duelling_partner
			
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3
					min = -49
				}
				scope:duelling_partner = {
					add_prestige = 100
				}
				add_stress = exertion_stress_cost
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3
					min = -49
				}
				add_prestige = 100
				scope:duelling_partner = {
					add_stress = exertion_stress_cost
				}
			}
		}
	}
	
	option = {
		name = camp.0300.sharpen_weapon
		
		add_character_modifier = {
			modifier = honed_blade 
			months = 3
		}
	}
	
	option = {
		name = camp.rest
		long_rest_effect = yes
	}
}

			}

				}
			}
		}
	}
	
	option = {
		name = camp.0400.lay_with_them
