namespace = festival_banquet

festival_banquet.0001 = { # Choose someone to give a toast
	type = activity_event
	title = "A Toast"
	desc = "Choose who you want to give the toast."
	
	theme = feast_activity
	left_portrait = {
		character = root
		animation = toast_goblet
	}
	center_portrait = {
		trigger = { exists = scope:chancellor }
		character = scope:chancellor
		animation = personality_honorable
	}
	right_portrait = {
		trigger = {exists = scope:my_spouse}
		character = scope:my_spouse
		animation = ecstasy
	}
	immediate = {
		involved_activity ?= {
			random_attending_character ?= {
				limit = {root.primary_spouse = this}
				save_scope_as = my_spouse
			}
			random_attending_character ?= {
				limit = {
					is_councillor_of = root
					has_council_position = councillor_chancellor
				}
				save_scope_as = chancellor
			}
		}
	}
	option = {
		name = "Myself, of course!"
		festival_challenge_toast = yes
		save_scope_as = toast_giver
		trigger_event = festival_banquet.0002
	}
	option = {
		name = "My chancellor."
		trigger = {exists = scope:chancellor}
		scope:chancellor ?= {
			festival_challenge_toast = yes
			save_scope_as = toast_giver
			trigger_event = festival_banquet.0002
		}
	}
	option = {
		name = "My spouse."
		trigger = { exists = scope:my_spouse }
		scope:my_spouse ?= {
			festival_challenge_toast = yes
			save_scope_as = toast_giver
			trigger_event = festival_banquet.0002
		}
	}
}

festival_banquet.0002 = { # Actually give the toast
	type = activity_event
	title = "A Toast"
	desc = "."
	
	theme = feast_activity
	left_portrait = {
		character = root
		animation = toast_goblet
	}
	
	option = {
		name = "Popular toast."
		if = {
			limit = {not = {this = scope:host}}
			add_character_modifier = {modifier = festival_popular_toast years = 5}
		}
		scope:host = {
			add_character_modifier = {modifier = festival_popular_toast years = 5}
		}
		scope:activity = {
			every_attending_character = {
				limit = {not = {this = root}}
				trigger_event = festival_banquet.0013
			}
		}
	}
	option = {
		name = "Personal toast."
		if = {
			limit = {not = {this = scope:host}}
			scope:host = {
				add_opinion = {
					target = prev
					modifier = grateful_opinion
					opinion = 25
				}
			}
		}
		scope:host = {
			add_prestige = 250
		}
		scope:activity = {
			every_attending_character = {
				limit = {not = {this = root}}
				trigger_event = festival_banquet.0013
			}
		}
	}
	option = {
		name = "Pious toast."
		if = {
			limit = {not = {this = scope:host}}
			add_piety = 250
		}
		scope:host = {
			add_piety = 250
		}
		scope:activity = {
			every_attending_character = {
				limit = {not = {this = root}}
				trigger_event = festival_banquet.0013
			}
		}
	}
}

festival_banquet.1100 = {
	type = activity_event
	title = "A secret revealed."
	desc = "."
	
	theme = feast_activity
	left_portrait = {
		character = root
	}
	right_portrait = {
		character = scope:secret_recipient
	}
	
	trigger = {
		any_secret = {
			exists = this
		}
	}
	
	immediate = {
		scope:activity = {
			random_attending_character = {
				limit = { feast_default_1008_secret_recipient_trigger = yes }
				weight = {
					base = 1
					interesting_feast_participant_modifier = yes
				}
				save_scope_as = secret_recipient
			}
		}
		random_secret = {
			limit = { NOT = { is_known_by = scope:secret_recipient } }
			save_scope_as = secret_to_reveal
		}
	}
}
