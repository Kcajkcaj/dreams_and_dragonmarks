five_year_everyone_pulse = {
	on_actions = {
		landless_choose_wizardry_type
		wizardry_progress_pulse
	}
}

landless_choose_wizardry_type = {
	trigger = {
		is_landed = no
		has_trait = mage
	}
	effect = {
		if = {
			limit = {
				not = {
					has_character_modifier = studying_abjuration
					has_character_modifier = studying_conjuration
					has_character_modifier = studying_divination
					has_character_modifier = studying_enchantment
					has_character_modifier = studying_evocation
					has_character_modifier = studying_illusion
					has_character_modifier = studying_necromancy
					has_character_modifier = studying_transmutation
				}
			}
			random_list = {
			10 = {add_character_modifier = studying_abjuration}
			10 = {add_character_modifier = studying_conjuration}
			10 = {add_character_modifier = studying_divination}
			10 = {add_character_modifier = studying_enchantment}
			10 = {add_character_modifier = studying_evocation}
			10 = {add_character_modifier = studying_illusion}
			10 = {add_character_modifier = studying_necromancy}
			10 = {add_character_modifier = studying_transmutation}
			}
		}
	}
}

wizardry_progress_pulse = {
	trigger = {
		is_landed = no
		has_trait = mage
	}
	effect = {
		if = {limit = {has_character_modifier = studying_abjuration} add_trait_xp = {trait = mage track = abjuration value = 10}}
		if = {limit = {has_character_modifier = studying_conjuration} add_trait_xp = {trait = mage track = conjuration value = 10}}
		if = {limit = {has_character_modifier = studying_divination} add_trait_xp = {trait = mage track = divination value = 10}}
		if = {limit = {has_character_modifier = studying_enchantment} add_trait_xp = {trait = mage track = enchantment value = 10}}
		if = {limit = {has_character_modifier = studying_evocation} add_trait_xp = {trait = mage track = evocation value = 10}}
		if = {limit = {has_character_modifier = studying_illusion} add_trait_xp = {trait = mage track = illusion value = 10}}
		if = {limit = {has_character_modifier = studying_necromancy} add_trait_xp = {trait = mage track = necromancy value = 10}}
		if = {limit = {has_character_modifier = studying_transmutation} add_trait_xp = {trait = mage track = transmutation value = 10}}
	}
}
