feudal_government_levies = {
	uses_opinion_of_liege = yes
	
	obligation_levels = {
		default = {
			levies = {
				scope:vassal = {
					if = {
						limit = {
							is_ai = yes
						}
						value = 0.3
						if = {
							limit = {
								scope:opinion_of_liege > 0
							}
							add =
							{
								value = scope:opinion_of_liege
								multiply = 0.00305
							}
						}
						else = {
							add =
							{
								value = scope:opinion_of_liege
								multiply = 0.004
							}
						}
					}
					else = { # is a player
						value = 0.25
					}
				}
			}
			min_levies = {
				scope:liege = {
					if = {
						limit = {
							has_realm_law = crown_authority_1
						}
						value = 0.05
					}
					else_if = {
						limit = {
							has_realm_law = crown_authority_2
						}
						value = 0.15
					}
					else_if = {
						limit = {
							has_realm_law = crown_authority_3
						}
						value = 0.25
					}
					else = {
						value = 0.0
					}
				}
			}
		}
	}
}
